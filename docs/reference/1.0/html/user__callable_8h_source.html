191 /*!
192  * \brief Create user-defined callable from the specified function
193  *
194  * Creates instance of the UserCallable object which invokes specified function (f) with parsed params according
195  * to the description (ad). For instance:
196  * ```c++
197  * MakeCallable(
198  * [](const std::string& str1, const std::string& str2) {
199  * return str1 + " " + str2;
200  * },
201  * ArgInfo{"str1"}, ArgInfo{"str2", false, "default"}
202  * );
203  * ```
204  * In this sample lambda function with two string params will be invoked from the jinja2 template and provided with
205  * the specified params. Each param is described by \ref ArgInfo structure. Result of the function will be converted
206  * and passed back to the jinja2 template.
207  *
208  * In case the function should accept extra positional args or extra named args this params should be described the
209  * following name.
210  * - Extra positional args. \ref ArgInfo should describe this param with name `*args`. Param of the function should
211  * has \ref ValuesList type
212  * - Extra named args. \ref ArgInfo should describe this param with name `**kwargs`. Param of the function should
213  * has \ref ValuesMap type
214  * - Current template context. \ref ArgInfo should describe this param with name `*context`. Param of the function should
215  * has \ref GenericMap type
216  *
217  * \param f Function which should be called
218  * \param ad Function param descriptors
219  *
220  * \returns Instance of the properly initialized \ref UserCallable structure
221  */
222 template<typename Fn, typename ... ArgDescr>
223 auto MakeCallable(Fn&& f, ArgDescr&& ... ad)
224 {
225  return UserCallable {
226  [=, fn = std::forward<Fn>(f)](const UserCallableParams& params) {
227  return detail::InvokeUserCallable(fn, params, ad...);
228  },
229  {ArgInfo(std::forward<ArgDescr>(ad))...}
230  };
231 }
