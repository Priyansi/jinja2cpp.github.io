21 /*!
22  * \brief Generic interface to filesystem handlers (loaders)
23  *
24  * This interface should be implemented in order to provide custom file system handler. Interface provides most-common methods which are called by
25  * the template environment to load the particular template. `OpenStream` methods return the unique pointer to the generic `istream` object implementation.
26  * So, the exact type (ex. `ifstream`, `istringstream` etc.) of input stream is unspecified. In order to delete stream object correctly returned pointer
27  * provide the custom deleter which should properly delete the stream object.
28  */
29 class IFilesystemHandler
30 {
31 public:
32  //! Destructor
33  virtual ~IFilesystemHandler() = default;
34 
35  /*!
36  * \brief Method is called to open the file with the specified name in 'narrow-char' mode.
37  *
38  * Method should return unique pointer to the std::istream object with custom deleter (\ref CharFileStreamPtr) . Deleter should properly delete pointee
39  * stream object.
40  *
41  * @param name Name of the file to open
42  * @return Opened stream object or empty pointer in case of any error
43  */
44  virtual CharFileStreamPtr OpenStream(const std::string& name) const = 0;
45  /*!
46  * \brief Method is called to open the file with the specified name in 'wide-char' mode.
47  *
48  * Method should return unique pointer to the std::wistream object with custom deleter (\ref WCharFileStreamPtr) . Deleter should properly delete pointee
49  * stream object.
50  *
51  * @param name Name of the file to open
52  * @return Opened stream object or empty pointer in case of any error
53  */
54  virtual WCharFileStreamPtr OpenWStream(const std::string& name) const = 0;
55  /*!
56  * \brief Method is called to obtain the modification date of the specified file (if applicable)
57  *
58  * If the underlaying filesystem supports retrival of the last modification date of the file this method should return this date when called. In other
59  * case it should return the empty optional object. Main purpose of this method is to help templates loader to determine the necessity of cached template
60  * reload
61  *
62  * @param name Name of the file to get the last modification date
63  * @return Last modification date (if applicable) or empty optional object otherwise
64  */
65  virtual nonstd::optional<std::chrono::system_clock::time_point> GetLastModificationDate(const std::string& name) const = 0;
